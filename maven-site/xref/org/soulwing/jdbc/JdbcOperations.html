20  package org.soulwing.jdbc;
21  
22  import org.soulwing.jdbc.source.SQLSource;
23  
24  /**
25   * A high level API for performing SQL operations using JDBC.
26   * <p>
27   * The design of this interface was heavily influenced by the
28   * {@code FluentJdbc} abstraction in the Spring Framework, however it
29   * differs significantly, particularly in the manner in which queries and
30   * updates are invoked.  Nonetheless, users of {@code FluentJdbc} should
31   * find it familiar enough to easily learn.
32   *
33   * @author Carl Harris
34   */
35  public interface JdbcOperations {
36  
37    /**
38     * Executes a single SQL (DDL) statement.
39     * @param sql the SQL statement to execute
40     */
41    void execute(String sql);
42  
43    /**
44     * Executes a single SQL (DDL) statement.
45     * @param source source for the SQL statement to execute
46     */
47    void execute(SQLSource source);
48  
49    /**
50     * Executes the sequence of SQL statements produced by the given source.
51     * <p>
52     * This method is a synonym for {@link #executeScript(SQLSource, boolean)}
53     * with the {@code ignoreErrors} parameter set to {@code false}.
54     *
55     * @param source source of the SQL statements to execute
56     */
57    void executeScript(SQLSource source);
58  
59    /**
60     * Executes the sequence of SQL statements produced by the given source.
61     * @param source source of the SQL statements to execute
62     * @param ignoreErrors flag indicating whether errors should be ignored
63     */
64    void executeScript(SQLSource source, boolean ignoreErrors);
65  
66    /**
67     * Creates a query operation.
68     * <p>
69     * This is a synonym for {@link #queryForType(Class)} with {@link Void}
70     * as the specified type.  A query obtained via this method is typically
71     * configured to use a {@link ResultSetHandler} to process returned rows.
72     *
73     * @return query object that can be configured and executed to retrieve
74     *    rows from the database
75     */
76    JdbcQuery<Void> query();
77  
78    /**
79     * Creates a query operation for objects of a given type.
80     * @param type subject type of the query
81     * @return a query that can be configured and executed to retreive rows
82     *    from the database and return objects of type {@code T}.
83     */
84    <T> JdbcQuery<T> queryForType(Class<T> type);
85  
86    /**
87     * Creates an update operation.
88     * @return an updater that can be configured and executed to update rows in
89     *    the database.
90     */
91    JdbcUpdate update();
92  
93    /**
94     * Creates a call operation.
95     * @param sql SQL call statement
96     * @return call operation
97     */
98    JdbcCall call(String sql);
99  
100   /**
101    * Creates a call operation.
102    * @param source source for the SQL call statement
103    * @return call operation;
104    */
105   JdbcCall call(SQLSource source);
106 
107 }
