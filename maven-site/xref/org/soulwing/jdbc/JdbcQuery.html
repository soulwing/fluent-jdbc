19  package org.soulwing.jdbc;
20  
21  import java.util.List;
22  
23  import org.soulwing.jdbc.source.SQLSource;
24  
25  /**
26   * An SQL query operation.
27   * <p>
28   * This API uses the builder pattern to allow a query to be easily configured
29   * and executed.  It supports queries that return many rows as well as those
30   * that return a single row.  The API is designed to allow a fluent specification
31   * of the query configuration.
32   * <p>
33   * An instance of this type is created using the {@link JdbcOperations#query query}
34   * or {@link JdbcOperations#queryForType(Class) queryForType} method on the
35   * {@link JdbcOperations} interface.  The user of this interface must configure
36   * at least two characteristics:
37   * <ol>
38   * <li> the SQL statement to execute, specified via the {@link #using(String) using}
39   *      method
40   * <li> the manner in which the results are to be handled; either
41   *      {@linkplain #mappingRowsWith(RowMapper) mapping rows to objects},
42   *      {@linkplain #extractingColumn(int) extracting a column value}, or using a
43   *      {@linkplain #handlingResultWith(ResultSetHandler) result set handler}
44   * </ol>
45   * <p>
46   * Additionally, the query can be configured for single execution (default)
47   * or repeated execution.  When a query is configured for repeated execution
48   * (using {@link #repeatedly()}), the underlying statement and connection
49   * objects remain open after a result is retrieved, allowing the query to be
50   * executed again with different parameter values.  A query that is configured
51   * for repeated execution must be closed when it is no longer needed, by
52   * invoking the {@link #close()} method explicitly or by enclosing it in a
53   * <em>try-with-resources</em> construct.
54   * <p>
55   * After a query has been fully configured, a result can be retrieved using
56   * either
57   * <ul>
58   * <li> {@link #retrieveList(Parameter...)} for a query that returns multiple
59   *      rows, OR
60   * <li> {@link #retrieveValue(Parameter...)} for a query that returns <em>exactly
61   *      one</em> row
62   * </ul>
63   * <p>
64   * Examples:
65   * <p>
66   * Retrieving a list of objects that correspond to the rows returned by
67   * a query using a {@link RowMapper}:
68   * <pre>
69   * {@code
70   * // A mapper that maps column values of a person row to a Person object
71   * RowMapper<Person> personMapper = new RowMapper<Person>() { ... };
72   *
73   * List<Person> results = sqlTemplate.queryForType(Person.class)
74   *     .using("SELECT * FROM person ORDER BY name")
75   *     .mappingRowsWith(personMapper)
76   *     .retrieveList();
77   * }</pre>
78   * <p>
79   * Retrieving a list of values for a given column for all matching rows:
80   * <pre>
81   * {@code
82   * List<String> names = sqlTemplate.queryForType(String.class)
83   *     .using("SELECT * FROM person ORDER BY name")
84   *     .extractingColumn("name")
85   *     .retrieveList();
86   * }</pre>
87   * <p>
88   * Retrieving a single object using a {@link RowMapper}:
89   * <pre>
90   * {@code
91   * // A mapper that maps column values of a person row to a Person object
92   * RowMapper<Person> personMapper = new RowMapper<Person>() { ... };
93   *
94   * long id = 3;  // ID of the person to retrieve
95   * Person person = sqlTemplate.queryForType(Person.class)
96   *     .using("SELECT * FROM person WHERE id = ?")
97   *     .mappingRowsWith(personMapper)
98   *     .retrieveValue(Parameter.with(id));
99   * }</pre>
100  * <p>
101  * Retrieving a single value of a column:
102  * <pre>
103  * {@code
104  * int averageAge = sqlTemplate.queryForType(int.class)
105  *     .using("SELECT AVG(age) FROM person")
106  *     .extractingColumn()
107  *     .retrieveValue();
108  * }</pre>
109  * <p>
110  * Repeatedly executing a query with different parameters:
111  * <pre>
112  * {@code
113  * try (JdbcQuery<String> query = sqlTemplate.queryForType(String.class)
114  *     .using("SELECT name FROM person WHERE name LIKE ?")
115  *     .extractingColumn()
116  *     .repeatedly()) {
117  *   System.out.format("matching names: %s",
118  *       query.retrieveList("%en%"));
119  *   System.out.format("matching names: %s",
120  *       query.retrieveList("%sh%"));
121  * }
122  * }</pre>
123  * <p>
124  * Processing the returned result set using a {@link ResultSetHandler}:
125  * <pre>
126  * {@code
127  * sqlTemplate.query()
128  *     .using("SELECT * FROM person ORDER BY id")
129  *     .handlingResultWith(new ResultSetMapper<Void>() {
130  *        public void handleResult(ResultSet rs) throws SQLException {
131  *          while (rs.next()) {
132  *            exporter.exportPerson(rs.getLong("id"), rs.getLong("name"));
133  *          }
134  *          return null;
135  *        }
136  *     })
137  *     .retrieveValue();
138  * }</pre>
139  *
140  * @author Carl Harris
141  */
142 public interface JdbcQuery<T> extends AutoCloseable {
143 
144   /**
145    * Configures this query to execute the given SQL statement.
146    * @param sql the SQL statement to execute
147    * @return this query
148    */
149   JdbcQuery<T> using(String sql);
150 
151   /**
152    * Configures this query to execute the given SQL statement.
153    * @param source source for the SQL statement to execute
154    * @return this query
155    */
156   JdbcQuery<T> using(SQLSource source);
157 
158   /**
159    * Configures this query to use the given result set handler.
160    * <p>
161    * Invoking this method replaces any existing configured row mapper or
162    * column extractor with the given result set handler.
163    *
164    * @param handler result set extractor
165    * @return this query
166    */
167   JdbcQuery<T> handlingResultWith(ResultSetHandler<T> handler);
168 
169   /**
170    * Configures this query to extract the value of the first column.
171    * <p>
172    * Invoking this method replaces any existing configured result set extractor
173    * or row mapper with the given column extractor.
174    *
175    * @return this query
176    */
177   JdbcQuery<T> extractingColumn();
178 
179   /**
180    * Configures this query to extract the value of the given column.
181    * <p>
182    * Invoking this method replaces any existing configured result set extractor
183    * or row mapper with the given column extractor.
184    *
185    * @param index column index (index values start at 1)
186    * @return this query
187    */
188   JdbcQuery<T> extractingColumn(int index);
189 
190   /**
191    * Configures this query to extract the value of the given column.
192    * <p>
193    * Invoking this method replaces any existing configured result set extractor
194    * or row mapper with the given column extractor.
195    *
196    * @param label column label
197    * @return this query
198    */
199   JdbcQuery<T> extractingColumn(String label);
200 
201   /**
202    * Configures this query to use the given row mapper.
203    * <p>
204    * Invoking this method replaces any existing configured result set extractor
205    * or row mapper with the given column extractor.
206    *
207    * @param rowMapper row mapper
208    * @return this query
209    */
210   JdbcQuery<T> mappingRowsWith(RowMapper<T> rowMapper);
211 
212   /**
213    * Configures this query for repeated execution.
214    * <p>
215    * A query that is configured as repeatable must be explicitly closed by
216    * the caller using {@link #close()} when no longer needed.
217    *
218    * @return this query
219    */
220   JdbcQuery<T> repeatedly();
221 
222   /**
223    * Executes the query, retrieving the list of values for all matching rows.
224    * @param parameters values for query placeholders
225    * @return list of values of type {@code T} that were extracted/mapped by
226    *   this query
227    */
228   List<T> retrieveList(Parameter... parameters);
229 
230   /**
231    * Executes the query, retrieving a value representing the matching row
232    * @param parameters values for query placeholders
233    * @return value of type {@code T} that was extracted/mapped by this query
234    * @throws SQLNoResultException if no row was matched by this query
235    * @throws SQLNonUniqueResultException if more than one row was matched by
236    *    this query
237    */
238   T retrieveValue(Parameter... parameters);
239 
240   /**
241    * Executes the query, processing the result set with the configured handler.
242    * <p>
243    * This method is a synonym for {@link #retrieveValue(Parameter...)}, but
244    * does not return the result produced by the query.  It is typically used
245    * when the query is configured with a {@link ResultSetHandler} that has a
246    * void return type.
247    *
248    * @param parameters values for query placeholders
249    */
250   void execute(Parameter... parameters);
251 
252   /**
253    * Closes the JDBC resources associated with this query.
254    * <p>
255    * After a query is closed, its retrieval methods may not be subsequently
256    * invoked.
257    */
258   void close();
259 
260 }
