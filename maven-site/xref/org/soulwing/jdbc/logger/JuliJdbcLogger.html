19  package org.soulwing.jdbc.logger;
20  
21  import java.util.logging.Level;
22  import java.util.logging.Logger;
23  
24  import org.soulwing.jdbc.Parameter;
25  
26  /**
27   * A {@link JdbcLogger} that delegates to {@link java.util.logging.Logger}.
28   * <p>
29   * SQL statements are logged at the {@code FINE} level.  Bound parameter values
30   * are logged at the {@code FINEST} level.
31   *
32   * @author Carl Harris
33   */
34  public class JuliJdbcLogger implements JdbcLogger {
35  
36    private final Logger logger;
37  
38    /**
39     * Constructs a new instance.
40     * @param logger the delegate logger
41     */
42    public JuliJdbcLogger(Logger logger) {
43      this.logger = logger;
44    }
45  
46    @Override
47    public void writeStatement(String sql) {
48      if (logger.isLoggable(Level.FINE)) {
49        logger.fine(sql);
50      }
51    }
52  
53    @Override
54    public void writeParameters(Parameter[] parameters) {
55      if (logger.isLoggable(Level.FINEST)) {
56        for (int index = 0, max = parameters.length; index < max; index++) {
57          logger.finest(parameters[index].toString(index));
58        }
59      }
60    }
61    
62  }
